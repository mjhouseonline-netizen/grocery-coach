    <script>
        let state = { items: [], budget: 100 };
        let scanningTargetId = null;
        
        // YOUR GPT LINK
        const MEAL_BUDDY_LINK = "https://chatgpt.com/g/g-68e8559d108c819190104c689cd732b1-meal-buddy";

        function init() {
            const saved = localStorage.getItem('groceryCoach_v3');
            if(saved) state = JSON.parse(saved);
            render();
        }
        function save() {
            localStorage.setItem('groceryCoach_v3', JSON.stringify(state));
            render();
        }

        // --- BARCODE SCANNER (QuaggaJS) ---
        function startScanner(targetInputId) {
            scanningTargetId = targetInputId;
            document.getElementById('scanner-container').style.display = 'block';
            
            Quagga.init({
                inputStream : {
                    name : "Live",
                    type : "LiveStream",
                    target: document.querySelector('#interactive')
                },
                decoder : {
                    readers : ["ean_reader", "upc_reader"] // Common barcodes
                }
            }, function(err) {
                if (err) { console.log(err); return }
                Quagga.start();
            });

            Quagga.onDetected(function(result) {
                const code = result.codeResult.code;
                document.getElementById(scanningTargetId).value = code; // Fill input
                stopScanner();
                if(scanningTargetId === 'new-item') {
                    alert("Scanned: " + code + "\n(Enter name manually)");
                }
            });
        }

        function stopScanner() {
            Quagga.stop();
            document.getElementById('scanner-container').style.display = 'none';
        }

        // --- APP LOGIC ---
        function addItem() {
            const input = document.getElementById('new-item');
            const name = input.value.trim();
            if(!name) return;
            state.items.push({ id: Date.now(), name, price: 0, status: 'planned' });
            input.value = '';
            save();
        }

        function toggleItemStatus(id) {
            const item = state.items.find(i => i.id === id);
            if(!item) return;
            if(item.status === 'planned') item.status = 'checked';
            else if(item.status === 'checked') item.status = 'planned';
            save();
        }
        
        function startShopping() {
            state.items.forEach(i => { if(i.status === 'checked') i.status = 'cart'; });
            save();
            switchTab('shop');
        }

        function updatePrice(id, price) {
            const item = state.items.find(i => i.id === id);
            if(item) {
                item.price = parseFloat(price) || 0;
                save(); 
            }
        }
        
        function deleteItem(id) {
            if(confirm("Remove item?")) {
                state.items = state.items.filter(i => i.id !== id);
                save();
            }
        }

        function editBudget() {
            const val = prompt("Set Budget:", state.budget);
            if(val) { state.budget = parseFloat(val); save(); }
        }
        
        function clearCart() {
            if(confirm("Clear all items?")) { state.items = []; save(); }
        }

        // --- AI LOGIC ---
        function launchMealBuddy() {
            const ingredients = state.items.map(i => i.name).join(", ");
            const diet = document.getElementById('diet-select').value;
            
            if(!ingredients) {
                alert("Add items to your list first!");
                return;
            }

            const prompt = `Hi Meal Buddy! I have these ingredients: [${ingredients}]. 
            Preferences: ${diet || 'None'}. 
            Create a 3-day meal plan with recipes using these items. Minimize waste.`;

            navigator.clipboard.writeText(prompt).then(() => {
                window.open(MEAL_BUDDY_LINK, '_blank');
                // Optional: Show toast/alert
            });
        }

        // --- RENDERERS ---
        function render() {
            renderPlan();
            renderShop();
            renderChef();
            updateBudgetUI();
        }

        function renderPlan() {
            const list = document.getElementById('planning-list');
            list.innerHTML = '';
            const planItems = state.items.filter(i => i.status === 'planned' || i.status === 'checked');
            const hasChecked = planItems.some(i => i.status === 'checked');
            
            if(planItems.length === 0) list.innerHTML = '<div class="text-center text-slate-300 font-bold py-8">List is empty</div>';

            planItems.forEach(item => {
                const isChecked = item.status === 'checked';
                list.innerHTML += `
                    <div class="flex items-center justify-between p-3 rounded-xl border ${isChecked ? 'bg-green-50 border-green-200' : 'bg-white border-slate-100'}" onclick="toggleItemStatus(${item.id})">
                        <div class="flex items-center gap-3">
                            <div class="w-6 h-6 rounded-full border-2 ${isChecked ? 'bg-green-50 border-green-500' : 'border-slate-200'} flex items-center justify-center">
                                ${isChecked ? '<i class="fas fa-check text-white text-xs"></i>' : ''}
                            </div>
                            <span class="font-bold ${isChecked ? 'text-slate-400 line-through' : 'text-slate-700'}">${item.name}</span>
                        </div>
                        <button onclick="event.stopPropagation(); deleteItem(${item.id})" class="text-slate-300 hover:text-red-400 px-2"><i class="fas fa-times"></i></button>
                    </div>
                `;
            });
            
            const shopBtn = document.getElementById('shop-btn');
            if(hasChecked) shopBtn.classList.remove('hidden'); else shopBtn.classList.add('hidden');
        }

        function renderShop() {
            const list = document.getElementById('cart-list');
            list.innerHTML = '';
            const cartItems = state.items.filter(i => i.status === 'cart');

            if(cartItems.length === 0) list.innerHTML = '<div class="text-center text-slate-300 font-bold py-8">Cart is empty.<br>Go to Plan tab to add items!</div>';

            cartItems.forEach(item => {
                list.innerHTML += `
                    <div class="flex items-center gap-3 p-3 bg-white rounded-xl border border-slate-100 shadow-sm">
                        <div class="flex-1 font-bold text-slate-700">${item.name}</div>
                        <div class="flex items-center bg-slate-50 rounded-lg px-2 border border-slate-200">
                            <span class="text-slate-400 text-sm font-bold mr-1">$</span>
                            <input type="number" step="0.01" class="bg-transparent w-16 py-2 text-right font-bold text-slate-800 focus:outline-none" 
                                value="${item.price || ''}" placeholder="0.00"
                                onchange="updatePrice(${item.id}, this.value)">
                        </div>
                        <button onclick="deleteItem(${item.id})" class="text-slate-300 hover:text-red-400 px-2"><i class="fas fa-times"></i></button>
                    </div>
                `;
            });
        }
        
        function renderChef() {
            const display = document.getElementById('pantry-list-display');
            const names = state.items.map(i => i.name);
            if(names.length > 0) {
                display.innerText = names.join(", ");
            } else {
                display.innerText = "Add items to your list to see them here!";
            }
        }

        function updateBudgetUI() {
            const spent = state.items.reduce((sum, i) => sum + (parseFloat(i.price) || 0), 0);
            const remaining = state.budget - spent;
            document.getElementById('spend-display').innerText = `$${spent.toFixed(2)}`;
            document.getElementById('remaining-display').innerText = `$${remaining.toFixed(2)}`;
            const banner = document.getElementById('budget-banner');
            if(remaining < 0) banner.classList.add('over'); else banner.classList.remove('over');
        }

        function handleEnter(e, cb) { if(e.key === 'Enter') cb(); }

        function switchTab(tab) {
            ['plan', 'shop', 'chef'].forEach(t => {
                document.getElementById(`view-${t}`).classList.add('hidden');
                document.getElementById(`nav-${t}`).classList.remove('active');
            });
            document.getElementById(`view-${tab}`).classList.remove('hidden');
            document.getElementById(`nav-${tab}`).classList.add('active');
        }

        init();
        
    </script>
